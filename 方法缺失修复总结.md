    if (monster == null || stage < 1) return;
    
    // 限制最大阶段
    if (stage > maxEvolutionStages) {
        stage = maxEvolutionStages;
    }
    
    // 应用进化效果
    effects.applyEvolutionEffects(monster, stage);
    
    // 记录进化

}
